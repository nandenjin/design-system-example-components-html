export function extractBody(html) {
  const match = html.match(/<body[^>]*>([\s\S]*?)<\/body>/i);
  return match ? match[1].trim() : html;
}

export function resolveImageUrls(html) {
  return html.replace(/\b(src|srcset)="([^"]+)"/g, (match, attr, value) => {
    const resolved = value
      .split(",")
      .map((part) => {
        const [url, descriptor] = part.trim().split(/\s+/);
        const filename = url.split("/").pop();
        const resolvedUrl = imageAssets[filename] ?? url;
        return descriptor ? `${resolvedUrl} ${descriptor}` : resolvedUrl;
      })
      .join(", ");
    return `${attr}="${resolved}"`;
  });
}

export function storybookUrl(name) {
  const slug = `components-${name}--docs`
    .toLowerCase()
    .replace(/\//g, "-")
    .replace(/\s+/g, "-");
  return `./?path=/docs/${slug}`;
}
